**转换脚本**：
```python
# scripts/convert_camplusplus.py
import torch
import onnx
from campp.model import CAMpp

# 加载模型
model = CAMpp(num_classes=1000)
model.load_state_dict(torch.load("camp++.pt"))
model.eval()

# 导出 ONNX
dummy_input = torch.randn(1, 80, 300)  # (batch, time, feat)
torch.onnx.export(
    model,
    dummy_input,
    "cam++.onnx",
    opset_version=14,
    input_names=["features"],
    output_names=["embeddings"],
    dynamic_axes={
        "features": {0: "batch_size", 1: "time"},
        "embeddings": {0: "batch_size"}
    }
)

# INT8 量化
from onnxruntime.quantization import quantize_dynamic
quantize_dynamic(
    "cam++.onnx",
    "cam++.int8.onnx",
    weight_type=QuantType.QInt8
)
```

**核心代码**：
```cpp
// services/voiceprint_service.cpp
class VoiceprintService {
public:
    bool Initialize(const std::string& model_path) {
        // 加载 ONNX 模型
        session_ = new Ort::Session(env_, model_path.c_str(),
                                    Ort::SessionOptions{nullptr});

        // 初始化特征提取器
        fbank_extractor_ = new KaldiFbank(80, 16000);

        // 加载注册用户向量
        LoadRegisteredUsers();

        return true;
    }

    std::string Verify(const std::string& user_id,
                      const std::vector<float>& audio) {
        // 1. 特征提取
        auto features = fbank_extractor_->Extract(audio);

        // 2. 模型推理
        auto embedding = RunInference(features);

        // 3. 向量比对
        auto registered = GetEmbedding(user_id);
        float score = CosineSimilarity(embedding, registered);

        // 4. 返回 JSON
        return FormatResponse(user_id, score);
    }

private:
    Ort::Env env_;
    Ort::Session* session_;
    KaldiFbank* fbank_extractor_;
};
```

#define VP_CMD_VERIFY    0x02
#define VP_CMD_IDENTIFY  0x03

#pragma pack(push, 1)
typedef struct {
    uint8_t cmd;
    uint32_t data_len;
    char user_id[64];
} vp_request_header_t;

typedef struct {
    uint32_t status;
    float score;
    char user_id[64];
} vp_response_t;

## ⚠️ 风险与应对

| 风险 | 影响 | 应对措施 |
|------|------|---------|
| GPU 交付延迟 | 云端进度 | 提前2周采购，备选云服务商 |
| MTK NPU 适配困难 | 端侧性能 | 并行准备 CPU 方案 |
