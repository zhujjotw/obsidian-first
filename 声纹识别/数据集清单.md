```
音频格式要求：

- 客厅、卧室等家庭场景
- 包含背景噪声（电视、空调等）
- 不同距离（1m, 2m, 3m）
```

## 🔄 数据预处理

### 预处理流程

```
┌──────────────┐    ┌──────────────┐    ┌──────────────┐
│  原始音频     │ -> │  格式转换     │ -> │  VAD检测     │
│  (各种格式)   │    │  (16k/16bit) │    │  (去静音)    │
└──────────────┘    └──────────────┘    └──────────────┘
                                               │
┌──────────────┐    ┌──────────────┐    ┌──────▼───────┐
│  特征缓存     │ <- │  特征提取     │ <- │  归一化      │
│  (.npy)      │    │  (Fbank)     │    │  (-20dB)     │
└──────────────┘    └──────────────┘    └──────────────┘
```

### 预处理脚本

```python
# scripts/preprocess_data.py
"""
数据预处理脚本
"""
import os
import argparse
import torchaudio
import torchaudio.transforms as T
from pathlib import Path
import torch
import numpy as np
from tqdm import tqdm

class AudioPreprocessor:
    def __init__(self, target_sr=16000, target_dBFS=-20):
        self.target_sr = target_sr
        self.resampler = T.Resample(orig_freq=48000, new_freq=target_sr)
        self.vad = T.Vad(sample_rate=target_sr)
        self.target_dBFS = target_dBFS

    def process(self, input_path, output_path):
        """处理单个音频文件"""
        try:
            # 1. 加载音频
            waveform, sr = torchaudio.load(input_path)

            # 2. 转换为单声道
            if waveform.shape[0] > 1:
                waveform = torch.mean(waveform, dim=0, keepdim=True)

            # 3. 重采样
            if sr != self.target_sr:
                waveform = self.resampler(waveform)

            # 4. VAD 去静音
            waveform = self.vad(waveform)

            # 5. 音量归一化
            rms = torch.sqrt(torch.mean(waveform ** 2))
            if rms > 0:
                gain = 10 ** ((self.target_dBFS - 20 * torch.log10(rms)) / 20)
                waveform = waveform * gain

            # 6. 限幅
            waveform = torch.clamp(waveform, -1.0, 1.0)

            # 7. 保存
            torchaudio.save(output_path, waveform, self.target_sr)
            return True

        except Exception as e:
            print(f"Error processing {input_path}: {e}")
            return False

def preprocess_dataset(input_dir, output_dir):
    """批量预处理数据集"""
    input_path = Path(input_dir)
    output_path = Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)

    processor = AudioPreprocessor()

    # 获取所有音频文件
    audio_files = list(input_path.rglob("*.wav")) + \
                  list(input_path.rglob("*.mp3")) + \
                  list(input_path.rglob("*.flac"))

    print(f"Found {len(audio_files)} audio files")

    # 处理
    for audio_file in tqdm(audio_files):
        rel_path = audio_file.relative_to(input_path)
        out_file = output_path / rel_path.with_suffix(".wav")
        out_file.parent.mkdir(parents=True, exist_ok=True)

        processor.process(str(audio_file), str(out_file))

    print(f"Preprocessed data saved to {output_dir}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True, help="Input directory")
    parser.add_argument("--output", required=True, help="Output directory")
    args = parser.parse_args()

    preprocess_dataset(args.input, args.output)
```

## 📊 数据增强

### 增强策略

| 增强方法 | 参数范围 | 用途 |
|---------|---------|------|
| 混响 (Reverb) | RT60: 0.1-0.8s | 模拟房间环境 |
| 背景噪声 | SNR: 0-20dB | 提升鲁棒性 |
| 速度扰动 | 0.85-1.15x | 数据增广 |
| 音量变化 | ±6dB | 模拟不同距离 |

### 增强代码

```python
# scripts/augment_audio.py
"""
音频增强
"""
import torch
import torchaudio.transforms as T
import numpy as np

class AudioAugmentor:
    def __init__(self):
        self.noise_files = self._load_noise_files()

    def add_reverb(self, waveform, rir=None):
        """添加混响"""
        if rir is None:
            # 使用简化的混响模型
            delay = int(torch.randint(100, 500, (1,)).item())
            decay = torch.rand(1).item() * 0.5
            reverb = torch.zeros_like(waveform)
            reverb[:, delay:] = waveform[:, :-delay] * decay
            return waveform + reverb
        else:
            # 使用 RIR 卷积
            return torch.nn.functional.conv1d(
                waveform.unsqueeze(1),
                rir.unsqueeze(0).unsqueeze(0)
            ).squeeze(1)

    def add_noise(self, waveform, snr_range=(5, 20)):
        """添加背景噪声"""
        noise_file = np.random.choice(self.noise_files)
        noise, _ = torchaudio.load(noise_file)

        # 调整长度
        if noise.shape[1] < waveform.shape[1]:
            noise = noise.repeat(1, waveform.shape[1] // noise.shape[1] + 1)
        noise = noise[:, :waveform.shape[1]]

        # 计算 SNR
        snr = np.random.uniform(*snr_range)
        signal_power = torch.mean(waveform ** 2)
        noise_power = torch.mean(noise ** 2)
        scale = torch.sqrt(signal_power / (noise_power * (10 ** (snr / 10))))

        return waveform + noise * scale

    def speed_perturb(self, waveform, factors=[0.85, 1.0, 1.15]):
        """速度扰动"""
        factor = np.random.choice(factors)
        return torch.nn.functional.interpolate(
            waveform.unsqueeze(1),
            scale_factor=factor,
            mode="linear"
        ).squeeze(1)

    def volume_change(self, waveform, db_range=(-6, 6)):
        """音量变化"""
        db_change = np.random.uniform(*db_range)
        gain = 10 ** (db_change / 20)
        return waveform * gain

    def augment(self, waveform):
        """随机组合增强"""
        if np.random.rand() > 0.5:
            waveform = self.add_reverb(waveform)
        if np.random.rand() > 0.5:
            waveform = self.add_noise(waveform)
        if np.random.rand() > 0.5:
            waveform = self.speed_perturb(waveform)
        if np.random.rand() > 0.5:
            waveform = self.volume_change(waveform)
        return waveform
```

## 📈 数据质量评估

### 评估指标

```python
# scripts/evaluate_data.py
"""
数据质量评估
"""
import torchaudio
import numpy as np
from pathlib import Path

def evaluate_dataset(data_dir):
    """评估数据集质量"""
    audio_files = list(Path(data_dir).rglob("*.wav"))

    stats = {
        "total_files": len(audio_files),
        "durations": [],
        "sample_rates": [],
        "snr_levels": [],
        "zero_crossing_rates": []
    }

    for audio_file in audio_files:
        waveform, sr = torchaudio.load(audio_file)

        # 时长
        duration = waveform.shape[1] / sr
        stats["durations"].append(duration)

        # 采样率
        stats["sample_rates"].append(sr)

        # 零交叉率
        zcr = np.mean(np.diff(np.sign(waveform.numpy())) != 0)
        stats["zero_crossing_rates"].append(zcr)

        # SNR 估算
        signal_power = torch.mean(waveform ** 2)
        noise_power = torch.mean(waveform[:, :int(0.1 * sr)] ** 2)  # 前100ms作为噪声
        if noise_power > 0:
            snr = 10 * torch.log10(signal_power / noise_power)
            stats["snr_levels"].append(snr.item())

    # 打印统计
    print(f"=== 数据集质量报告 ===")
    print(f"总文件数: {stats['total_files']}")
    print(f"平均时长: {np.mean(stats['durations']):.2f}s")
    print(f"时长范围: {np.min(stats['durations']):.2f}s - {np.max(stats['durations']):.2f}s")
    print(f"平均 SNR: {np.mean(stats['snr_levels']):.2f} dB")
    print(f"平均零交叉率: {np.mean(stats['zero_crossing_rates']):.4f}")

    return stats
```
