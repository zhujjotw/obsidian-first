#### 2.1.2 数据预处理流程

```python
# scripts/data_preprocessing.py
"""
数据预处理流程：
1. 音频格式统一 (16kHz, 16bit, WAV)
2. VAD (Voice Activity Detection) 静音去除
3. 音量归一化 (-20dB)
4. 数据增强 (Reverb, Noise, Speed)
5. 特征提取 (Fbank 80-dim)
"""

import torch
import torchaudio
from torch import nn
import torch.nn.functional as F
import torchaudio.transforms as T

class AudioPreprocessor:
    def __init__(self, sample_rate=16000):
        self.sample_rate = sample_rate
        self.vad = T.Vad(sample_rate)
        self.resampler = T.Resample(orig_freq=48000, new_freq=16000)
        self.melspec = T.MelSpectrogram(
            sample_rate=16000,
            n_fft=512,
            win_length=400,
            hop_length=160,
            n_mels=80
        )

    def process(self, audio_path):
        # 1. 加载音频
        waveform, sr = torchaudio.load(audio_path)

        # 2. 重采样
        if sr != self.sample_rate:
            waveform = self.resampler(waveform)

        # 3. VAD 静音去除
        waveform = self.vad(waveform)

        # 4. 归一化
        waveform = waveform / (torch.max(torch.abs(waveform)) + 1e-8)

        # 5. 提取 Fbank 特征
        features = self.melspec(waveform)
        features = torch.log(features + 1e-8)

        return features
```

"""
模型优化与量化

import torch
from transformers import Wav2Vec2Model

def quantize_model(model_path, output_path):
    """INT8 量化，减少显存占用和推理延迟"""

    # 加载模型
    model = WavLMSpeakerEmbedding.load_from_checkpoint(model_path)
    model.eval()

    # 动态量化
    quantized_model = torch.quantization.quantize_dynamic(
        model,
        {torch.nn.Linear},
        dtype=torch.qint8
    )

    # 保存量化模型
    torch.save(quantized_model.state_dict(), output_path)

    print(f"Quantized model saved to {output_path}")

    # 对比模型大小
    original_size = os.path.getsize(model_path) / 1024 / 1024
    quantized_size = os.path.getsize(output_path) / 1024 / 1024

    print(f"Original size: {original_size:.2f} MB")
    print(f"Quantized size: {quantized_size:.2f} MB")
    print(f"Compression ratio: {original_size / quantized_size:.2f}x")

    return quantized_model

def export_to_onnx(model_path, output_path):
    """导出 ONNX 格式，用于端侧部署"""

    model = WavLMSpeakerEmbedding.load_from_checkpoint(model_path)
    model.eval()

    dummy_input = torch.randn(1, 16000 * 5)  # 5秒音频

        model,
        dummy_input,

        dynamic_axes={
            'audio': {0: 'batch_size'},
            'embedding': {0: 'batch_size'}
        }

# app/api/enroll.py
"""
声纹注册接口
"""
from fastapi import APIRouter, UploadFile, File, HTTPException, Depends
from pydantic import BaseModel
from typing import Optional
import numpy as np

from app.services.wavlm_service import wavlm_service
from app.services.vector_store import vector_store
from app.services.liveness import liveness_detector
from app.core.security import verify_api_key

router = APIRouter()

class EnrollRequest(BaseModel):
    user_id: str
    device_id: Optional[str] = None

class EnrollResponse(BaseModel):
    status: str
    user_id: str
    enrollment_id: str
    message: str

@router.post("/enroll", response_model=EnrollResponse)
async def enroll_speaker(
    user_id: str,
    audio: UploadFile = File(...),
    device_id: Optional[str] = None,
    api_key: str = Depends(verify_api_key)
):
    """
    注册用户声纹

    参数:
    - user_id: 用户唯一标识
    - audio: 音频文件 (WAV/MP3, 最多10秒)
    - device_id: 设备标识 (可选)

    返回:
    - enrollment_id: 注册ID
    - status: 注册状态
    """
    try:
        # 1. 验证音频格式
        if audio.content_type not in ["audio/wav", "audio/mpeg", "audio/x-wav"]:
            raise HTTPException(status_code=400, detail="不支持的音频格式")

        # 2. 读取音频数据
        audio_bytes = await audio.read()
        audio_array = np.frombuffer(audio_bytes, dtype=np.int16)
        audio_array = audio_array.astype(np.float32) / 32768.0  # 归一化

        # 3. 活体检测
        is_live, spoof_score = await liveness_detector.detect(audio_array)
        if not is_live:
            raise HTTPException(status_code=400, detail="活体检测失败，可能是录音攻击")

        # 4. 提取声纹特征
        embedding = await wavlm_service.extract_embedding(audio_array)

        # 5. 检查是否已注册
        existing = await vector_store.get_user_embedding(user_id)
        if existing is not None:
            # 更新现有声纹
            enrollment_id = await vector_store.update_embedding(user_id, embedding)
        else:
            # 新注册
            enrollment_id = await vector_store.store_embedding(user_id, embedding, device_id)

        return EnrollResponse(
            status="success",
            user_id=user_id,
            enrollment_id=enrollment_id,
            message="声纹注册成功"
        )

    except Exception as e:
        raise HTTPException(status_code=500, detail=f"注册失败: {str(e)}")

    - is_matched: 是否匹配
    - score: 相似度分数 (0-1)
    - confidence: 置信度等级
    """
    import uuid
    request_id = f"vprint_{uuid.uuid4().hex[:12]}"

    try:
        # 1. 读取音频
        audio_bytes = await audio.read()
        audio_array = np.frombuffer(audio_bytes, dtype=np.int16)
        audio_array = audio_array.astype(np.float32) / 32768.0

        # 2. 活体检测
        is_live, spoof_score = await liveness_detector.detect(audio_array)
        if not is_live:
            return VerifyResponse(
                status="spoof_detected",
                request_id=request_id,
                recognition_result={
                    "speaker_id": user_id,
                    "score": 0.0,
                    "is_matched": False
                },
                confidence="attack_detected"
            )

        # 3. 提取声纹特征
        embedding = await wavlm_service.extract_embedding(audio_array)

        # 4. 获取注册声纹
        registered_embedding = await vector_store.get_user_embedding(user_id)
        if registered_embedding is None:
            raise HTTPException(status_code=404, detail=f"用户 {user_id} 未注册声纹")

        # 5. 计算相似度
        score = cosine_similarity(embedding, registered_embedding)

        # 6. 判断是否匹配
        is_matched = score >= threshold

        # 7. 确定置信度等级
        if score >= 0.95:
            confidence = "high"
        elif score >= 0.85:
            confidence = "medium"
        else:
            confidence = "low"

        return VerifyResponse(
            status="success",
            request_id=request_id,
            recognition_result={
                "speaker_id": user_id,
                "score": float(score),
                "confidence": confidence,
                "is_matched": is_matched
            },
            confidence=confidence
        )

    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"验证失败: {str(e)}")


def cosine_similarity(a, b):
    """计算余弦相似度"""
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b) + 1e-8)

# app/api/identify.py
"""
声纹辨识接口 (1:N Identification)
"""
from fastapi import APIRouter, UploadFile, File, HTTPException, Depends
from pydantic import BaseModel
from typing import Optional, List
import numpy as np

from app.services.wavlm_service import wavlm_service
from app.services.vector_store import vector_store
from app.services.liveness import liveness_detector
from app.core.security import verify_api_key

router = APIRouter()

class IdentifyResponse(BaseModel):
    status: str
    request_id: str
    recognition_result: dict
    top_candidates: Optional[List[dict]] = None

@router.post("/identify", response_model=IdentifyResponse)
async def identify_speaker(
    audio: UploadFile = File(...),
    top_k: Optional[int] = 5,
    threshold: Optional[float] = 0.80,
    device_id: Optional[str] = None,
    api_key: str = Depends(verify_api_key)
):
    """
    1:N 声纹辨识

    从所有已注册用户中识别说话人

    参数:
    - audio: 音频文件
    - top_k: 返回前K个候选结果
    - threshold: 最低相似度阈值
    - device_id: 设备标识 (用于过滤该设备注册的用户)

    返回:
    - speaker_id: 识别出的用户ID
    - score: 相似度分数
    - top_candidates: 候选列表
    """
    import uuid
    request_id = f"vprint_{uuid.uuid4().hex[:12]}"

    try:
        # 1. 读取音频
        audio_bytes = await audio.read()
        audio_array = np.frombuffer(audio_bytes, dtype=np.int16)
        audio_array = audio_array.astype(np.float32) / 32768.0

        # 2. 活体检测
        is_live, spoof_score = await liveness_detector.detect(audio_array)
        liveness_info = {
            "is_live": is_live,
            "spoof_score": float(spoof_score),
            "attack_type": "none" if is_live else "replay"
        }

        if not is_live:
            return IdentifyResponse(
                status="spoof_detected",
                request_id=request_id,
                recognition_result={
                    "speaker_id": None,
                    "score": 0.0,
                    "is_matched": False,
                    "liveness_check": liveness_info
                }
            )

        # 3. 提取声纹特征
        embedding = await wavlm_service.extract_embedding(audio_array)

        # 4. 向量检索 (1:N)
        candidates = await vector_store.search(
            embedding,
            top_k=top_k,
            device_id=device_id
        )

        # 5. 过滤低于阈值的候选
        valid_candidates = [
            c for c in candidates
            if c['score'] >= threshold
        ]

        if not valid_candidates:
            return IdentifyResponse(
                status="no_match",
                request_id=request_id,
                recognition_result={
                    "speaker_id": None,
                    "score": 0.0,
                    "is_matched": False,
                    "liveness_check": liveness_info
                },
                top_candidates=[]
            )

        # 6. 返回最佳匹配
        best_match = valid_candidates[0]

        return IdentifyResponse(
            status="success",
            request_id=request_id,
            recognition_result={
                "speaker_id": best_match['user_id'],
                "score": float(best_match['score']),
                "confidence": "high" if best_match['score'] >= 0.95 else "medium",
                "is_matched": True,
                "liveness_check": liveness_info
            },
            top_candidates=[
                {
                    "user_id": c['user_id'],
                    "score": float(c['score'])
                }
                for c in valid_candidates[:top_k]
            ]
        )

    except Exception as e:
        raise HTTPException(status_code=500, detail=f"辨识失败: {str(e)}")

# app/services/wavlm_service.py
"""
WavLM 声纹特征提取服务
"""
import torch
import numpy as np
from transformers import AutoModel, AutoFeatureExtractor
import logging

logger = logging.getLogger(__name__)

class WavLMService:
    def __init__(self, model_name="microsoft/wavlm-base-plus", device=None):
        self.model_name = model_name
        self.device = device or ("cuda" if torch.cuda.is_available() else "cpu")
        self.model = None
        self.feature_extractor = None
        self._initialized = False

    async def initialize(self):
        """延迟初始化模型"""
        if self._initialized:
            return

        logger.info(f"Loading WavLM model: {self.model_name}")

        # 加载模型
        self.model = AutoModel.from_pretrained(self.model_name)
        self.feature_extractor = AutoFeatureExtractor.from_pretrained(self.model_name)

        # 微调后的权重
        checkpoint_path = "models/checkpoints/best_model_eer_0.0350.pt"
        try:
            checkpoint = torch.load(checkpoint_path, map_location=self.device)
            self.model.load_state_dict(checkpoint['model_state_dict'])
            logger.info(f"Loaded fine-tuned weights from {checkpoint_path}")
        except FileNotFoundError:
            logger.warning(f"Fine-tuned weights not found, using pre-trained model")

        self.model.to(self.device)
        self.model.eval()
        self._initialized = True

        # 预热
        dummy_audio = torch.randn(1, 16000).to(self.device)
        with torch.no_grad():
            _ = self.model(dummy_audio)

        logger.info("WavLM model initialized successfully")

    async def extract_embedding(self, audio_array: np.ndarray) -> np.ndarray:
        """
        提取声纹特征向量

        参数:
            audio_array: 音频数据 (float32, 范围 [-1, 1], 采样率 16000)

        返回:
            embedding: 768维特征向量
        """
        if not self._initialized:
            await self.initialize()

        # 确保音频长度
        if len(audio_array) < 16000:
            # 填充
            padding = np.zeros(16000 - len(audio_array))
            audio_array = np.concatenate([audio_array, padding])
        elif len(audio_array) > 16000 * 10:
            # 截断最多10秒
            audio_array = audio_array[:16000 * 10]

        # 转换为tensor
        audio_tensor = torch.from_numpy(audio_array).unsqueeze(0).to(self.device)

        # 特征提取
        inputs = self.feature_extractor(
            audio_array,
            sampling_rate=16000,
            return_tensors="pt"
        ).to(self.device)

        # 推理
        with torch.no_grad():
            outputs = self.model(**inputs)
            # 使用最后一层的平均池化作为句子级嵌入
            embeddings = outputs.last_hidden_state.mean(dim=1)

        # L2 归一化
        embeddings = torch.nn.functional.normalize(embeddings, p=2, dim=1)

        return embeddings.cpu().numpy()[0]

    async def extract_embedding_batch(self, audio_list: list) -> np.ndarray:
        """批量提取特征"""
        if not self._initialized:
            await self.initialize()

        embeddings = []
        for audio_array in audio_list:
            emb = await self.extract_embedding(audio_array)
            embeddings.append(emb)

        return np.array(embeddings)

# app/services/vector_store.py
"""
向量存储服务 (Milvus/Redis)
"""
import redis
import numpy as np
import json
from typing import Optional, List, Dict
from datetime import datetime
import logging

logger = logging.getLogger(__name__)

class VectorStore:
    def __init__(self, redis_url="redis://localhost:6379/0"):
        self.redis = redis.from_url(redis_url)
        self.embedding_dim = 768

    async def store_embedding(
        self,
        user_id: str,
        embedding: np.ndarray,
        device_id: Optional[str] = None
    ) -> str:
        """存储用户声纹向量"""
        import uuid
        enrollment_id = f"enroll_{user_id}_{uuid.uuid4().hex[:8]}"
        timestamp = datetime.utcnow().isoformat()

        key = f"voiceprint:{user_id}"
        value = {
            "embedding": embedding.tolist(),
            "user_id": user_id,
            "device_id": device_id,
            "enrollment_id": enrollment_id,
            "created_at": timestamp,
            "updated_at": timestamp
        }

        self.redis.setex(
            key,
            86400 * 365,  # 1年过期
            json.dumps(value)
        )

        # 同时添加到索引集合
        self.redis.sadd("voiceprint:all_users", user_id)
        if device_id:
            self.redis.sadd(f"voiceprint:device:{device_id}", user_id)

        logger.info(f"Stored embedding for user {user_id}")
        return enrollment_id

    async def get_user_embedding(self, user_id: str) -> Optional[np.ndarray]:
        """获取用户声纹向量"""
        key = f"voiceprint:{user_id}"
        data = self.redis.get(key)

        if data is None:
            return None

        value = json.loads(data)
        return np.array(value["embedding"])

    async def update_embedding(self, user_id: str, new_embedding: np.ndarray) -> str:
        """更新用户声纹向量"""
        key = f"voiceprint:{user_id}"
        existing_data = self.redis.get(key)

        if existing_data:
            value = json.loads(existing_data)
            value["embedding"] = new_embedding.tolist()
            value["updated_at"] = datetime.utcnow().isoformat()

            self.redis.setex(
                key,
                86400 * 365,
                json.dumps(value)
            )
            return value["enrollment_id"]

        return await self.store_embedding(user_id, new_embedding)

    async def search(
        self,
        query_embedding: np.ndarray,
        top_k: int = 5,
        device_id: Optional[str] = None
    ) -> List[Dict]:
        """
        搜索最相似的声纹

        返回:
            List[Dict]: [{"user_id": str, "score": float}, ...]
        """
        # 获取候选用户集合
        if device_id:
            user_ids = list(self.redis.smembers(f"voiceprint:device:{device_id}"))
        else:
            user_ids = list(self.redis.smembers("voiceprint:all_users"))

        # 批量获取嵌入向量
        candidates = []
        for uid in user_ids:
            uid = uid.decode() if isinstance(uid, bytes) else uid
            embedding = await self.get_user_embedding(uid)
            if embedding is not None:
                score = cosine_similarity(query_embedding, embedding)
                candidates.append({
                    "user_id": uid,
                    "score": score
                })

        # 按分数排序
        candidates.sort(key=lambda x: x["score"], reverse=True)

        return candidates[:top_k]

    async def delete_user(self, user_id: str) -> bool:
        """删除用户声纹"""
        key = f"voiceprint:{user_id}"
        result = self.redis.delete(key)
        self.redis.srem("voiceprint:all_users", user_id)
        return result > 0

    async def get_stats(self) -> Dict:
        """获取统计信息"""
        total_users = self.redis.scard("voiceprint:all_users")
        return {
            "total_enrollments": total_users,
            "embedding_dim": self.embedding_dim
        }


def cosine_similarity(a, b):
    """计算余弦相似度"""
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b) + 1e-8)

# app/services/liveness.py
"""
活体检测服务
检测录音回放攻击
"""
import numpy as np
import logging

logger = logging.getLogger(__name__)

class LivenessDetector:
    def __init__(self):
        self.spoof_threshold = 0.5

    async def detect(self, audio_array: np.ndarray) -> tuple:
        """
        活体检测

        返回:
            (is_live, spoof_score)
            - is_live: True 为真人, False 为攻击
            - spoof_score: 攻击概率分数
        """
        # 1. 检查音频质量指标
        score = 0.0

        # 1.1 能量检测 (过低可能是环境噪声)
        energy = np.mean(audio_array ** 2)
        if energy < 0.001:
            return False, 0.9  # 高概率是攻击

        # 1.2 零交叉率 (录音通常较低)
        zero_crossings = np.sum(np.diff(np.sign(audio_array)) != 0)
        zcr = zero_crossings / len(audio_array)
        if zcr < 0.05:
            score += 0.3

        # 1.3 频谱分析 (录音缺少高频)
        fft = np.fft.fft(audio_array)
        magnitude = np.abs(fft[:len(fft)//2])
        high_freq_energy = np.sum(magnitude[len(magnitude)//2:])
        total_energy = np.sum(magnitude) + 1e-8
        if high_freq_energy / total_energy < 0.1:
            score += 0.4

        # 2. 判断
        is_live = score < self.spoof_threshold
        spoof_score = min(score / self.spoof_threshold, 1.0)

        return is_live, spoof_score
